const express = require('express'); const cors = require('cors'); const fetch = require('node-fetch'); require('dotenv').config(); const app = express(); const PORT = process.env.PORT || 3001; const ALLOWED_ORIGINS = (process.env.ALLOWED_ORIGINS || 'http://localhost:5500,http://127.0.0.1:5500').split(','); app.use(cors({ origin: (origin, cb) => { if (!origin || ALLOWED_ORIGINS.includes(origin)) cb(null, true); else cb(new Error('Origem não permitida')); } })); app.use(express.json()); // ── Config SSótica (URLs reais conforme documentação oficial) ────────────── const SS_BASE = 'https://app.ssotica.com.br/api/v1'; const SS_KEY = process.env.SSOTICA_API_KEY; const SS_CNPJ = process.env.SSOTICA_CNPJ; // CNPJ da loja sem pontuação if (!SS_KEY) { console.error('❌ SSOTICA_API_KEY não definida no .env'); process.exit(1); } // Helper: chamada autenticada à SSótica async function ssotica(path, params = {}) { const url = new URL(`${SS_BASE}${path}`); Object.entries(params).forEach(([k, v]) => { if (v !== undefined) url.searchParams.set(k, v); }); const res = await fetch(url.toString(), { headers: { 'Authorization': `Bearer ${SS_KEY}`, 'Content-Type': 'application/json' } }); const text = await res.text(); let data; try { data = JSON.parse(text); } catch { data = { raw: text }; } return { status: res.status, data }; } // Datas auxiliares function hoje() { return new Date().toISOString().split('T')[0]; } function diasAtras(n) { const d = new Date(); d.setDate(d.getDate() - n); return d.toISOString().split('T')[0]; } // ── Status ───────────────────────────────────────────────────────────────── app.get('/api/status', async (req, res) => { try { // Testa com uma busca de vendas do dia atual const cnpj = SS_CNPJ || req.query.cnpj; if (!cnpj) return res.json({ ok: false, msg: 'SSOTICA_CNPJ não configurado', ts: new Date().toISOString() }); const r = await ssotica('/integracoes/vendas/periodo', { cnpj, inicio_periodo: hoje(), fim_periodo: hoje() }); res.json({ ok: r.status === 200, status_http: r.status, ts: new Date().toISOString() }); } catch (e) { res.json({ ok: false, erro: e.message, ts: new Date().toISOString() }); } }); // ── Vendas por período ────────────────────────────────────────────────────── // GET /api/vendas?inicio=2025-01-01&fim=2025-01-31&page=1&perPage=100 app.get('/api/vendas', async (req, res) => { try { const cnpj = SS_CNPJ || req.query.cnpj; const inicio = req.query.inicio || diasAtras(30); const fim = req.query.fim || hoje(); const page = req.query.page || 1; const perPage = req.query.perPage || 100; const { status, data } = await ssotica('/integracoes/vendas/periodo', { cnpj, inicio_periodo: inicio, fim_periodo: fim, page, perPage }); res.status(status).json(data); } catch (e) { res.status(500).json({ erro: e.message }); } }); // ── Ordens de Serviço por período ────────────────────────────────────────── // GET /api/ordens?inicio=2025-01-01&fim=2025-01-31 app.get('/api/ordens', async (req, res) => { try { const cnpj = SS_CNPJ || req.query.cnpj; const inicio = req.query.inicio || diasAtras(30); const fim = req.query.fim || hoje(); const page = req.query.page || 1; const perPage = req.query.perPage || 100; const { status, data } = await ssotica('/integracoes/ordens-servico/periodo', { cnpj, inicio_periodo: inicio, fim_periodo: fim, page, perPage }); res.status(status).json(data); } catch (e) { res.status(500).json({ erro: e.message }); } }); // ── Financeiro / Extrato por período ─────────────────────────────────────── app.get('/api/financeiro', async (req, res) => { try { const cnpj = SS_CNPJ || req.query.cnpj; const inicio = req.query.inicio || diasAtras(30); const fim = req.query.fim || hoje(); const { status, data } = await ssotica('/integracoes/financeiro/extrato/periodo', { cnpj, inicio_periodo: inicio, fim_periodo: fim }); res.status(status).json(data); } catch (e) { res.status(500).json({ erro: e.message }); } }); // ── Contas a Receber ──────────────────────────────────────────────────────── app.get('/api/contas-receber', async (req, res) => { try { const empresa = SS_CNPJ || req.query.empresa; const inicio = req.query.inicio || diasAtras(30); const fim = req.query.fim || hoje(); const { status, data } = await ssotica('/integracoes/financeiro/contas-a-receber/periodo', { empresa, inicio_periodo: inicio, fim_periodo: fim, tipo_periodo: req.query.tipo || 'vencimento', page: req.query.page || 1, perPage: req.query.perPage || 100 }); res.status(status).json(data); } catch (e) { res.status(500).json({ erro: e.message }); } }); // ── Contas a Pagar ────────────────────────────────────────────────────────── app.get('/api/contas-pagar', async (req, res) => { try { const empresa = SS_CNPJ || req.query.empresa; const inicio = req.query.inicio || diasAtras(30); const fim = req.query.fim || hoje(); const { status, data } = await ssotica('/integracoes/financeiro/contas-a-pagar/periodo', { empresa, inicio_periodo: inicio, fim_periodo: fim, tipo_periodo: req.query.tipo || 'vencimento', page: req.query.page || 1, perPage: req.query.perPage || 100 }); res.status(status).json(data); } catch (e) { res.status(500).json({ erro: e.message }); } }); // ── Estoque / Produtos ────────────────────────────────────────────────────── app.get('/api/produtos', async (req, res) => { try { const empresa = SS_CNPJ || req.query.empresa; const { status, data } = await ssotica('/produto/estoque/busca', { empresa, referencia: req.query.referencia, id: req.query.id, page: req.query.page || 1, perPage: req.query.perPage || 100 }); res.status(status).json(data); } catch (e) { res.status(500).json({ erro: e.message }); } }); // ── Pós-Venda: vendas dos últimos 400 dias classificadas por etapa ───────── app.get('/api/pos-venda', async (req, res) => { try { const cnpj = SS_CNPJ || req.query.cnpj; const hoje_dt = new Date(); // Busca vendas dos últimos 400 dias em páginas de 100 let todasVendas = []; for (let p = 1; p <= 6; p++) { const { data } = await ssotica('/integracoes/vendas/periodo', { cnpj, inicio_periodo: diasAtras(400), fim_periodo: hoje(), page: p, perPage: 100 }); const lista = Array.isArray(data) ? data : (data.data || data.vendas || []); if (!Array.isArray(lista) || lista.length === 0) break; todasVendas = todasVendas.concat(lista); if (lista.length < 100) break; } const etapas = { dias15: [], mes1: [], dias75: [], meses6: [], meses8: [], meses10: [], meses11: [], meses12: [] }; todasVendas.forEach(v => { // Campos reais da SSótica conforme documentado if (!v.data) return; const dias = Math.floor((hoje_dt - new Date(v.data)) / 86400000); const primeiroItem = v.itens && v.itens[0]; const produto = (primeiroItem && primeiroItem.produto && primeiroItem.produto.descricao) || '—'; const extraItens = v.itens && v.itens.length > 1 ? ` +${v.itens.length - 1}` : ''; const item = { id: v.id, numero: v.numero, nome: (v.cliente && v.cliente.nome) || '—', cpf: (v.cliente && v.cliente.cpf_cnpj) || '', telefone: (v.cliente && v.cliente.telefones && v.cliente.telefones[0] && v.cliente.telefones[0].numero) || '', produto: produto + extraItens, dataVenda: v.data, valor: v.valor_liquido, diasDesdeCompra: dias, responsavel: (v.funcionario && v.funcionario.nome) || '—' }; if (dias >= 10 && dias <= 20) etapas.dias15.push(item); else if (dias >= 25 && dias <= 40) etapas.mes1.push(item); else if (dias >= 68 && dias <= 85) etapas.dias75.push(item); else if (dias >= 165 && dias <= 195) etapas.meses6.push(item); else if (dias >= 225 && dias <= 255) etapas.meses8.push(item); else if (dias >= 285 && dias <= 315) etapas.meses10.push(item); else if (dias >= 315 && dias <= 345) etapas.meses11.push(item); else if (dias >= 345 && dias <= 400) etapas.meses12.push(item); }); const total = Object.values(etapas).reduce((a, b) => a + b.length, 0); res.json({ etapas, totalPosvenda: total, totalVendas: todasVendas.length }); } catch (e) { res.status(500).json({ erro: e.message }); } }); app.listen(PORT, () => { console.log(`✅ ORO CRM Backend rodando em http://localhost:${PORT}`); console.log(`🔑 SSótica API Key: ****${SS_KEY ? SS_KEY.slice(-6) : 'NÃO DEFINIDA'}`); console.log(`🏪 CNPJ configurado: ${SS_CNPJ || 'NÃO DEFINIDO — passe via ?cnpj= ou configure SSOTICA_CNPJ'}`); });
