Como Integrar a SSótica API
Guia técnico — o que fazer quando receber as credenciais
Quando receber sua API Key da SSótica, substitua SUA_API_KEY_AQUI pela chave real e o sistema já começa a trazer dados reais automaticamente.
1. Buscar Dados do Cliente por CPF
const API_BASE = "https://api.ssotica.com.br/v1"; const API_KEY = "SUA_API_KEY_AQUI"; // ← substituir ao receber async function buscarCliente(cpf) { const res = await fetch(`${API_BASE}/clientes/${cpf}`, { headers: { "Authorization": `Bearer ${API_KEY}` } }); return res.json(); // { nome, cpf, telefone, email, cidade, ... } }
2. Receitas e Exames de Vista
async function buscarReceitas(cpf) { const res = await fetch(`${API_BASE}/clientes/${cpf}/receitas`, { headers: { "Authorization": `Bearer ${API_KEY}` } }); const data = await res.json(); // data.receitas = [{ data, od, oe, cilindro, eixo, add, validade, medico }] const ultima = data.receitas[0]; const vencida = new Date(ultima.validade) < new Date(); return { ...ultima, vencida }; } // Listar TODOS com receita vencida de uma vez async function receitasVencidas() { const res = await fetch(`${API_BASE}/receitas/vencidas`, { headers: { "Authorization": `Bearer ${API_KEY}` } }); return res.json(); // lista automática — sem checar 1 a 1 }
3. Histórico de Compras + Fase CRM Automática
async function buscarCompras(cpf) { const res = await fetch(`${API_BASE}/clientes/${cpf}/compras`, { headers: { "Authorization": `Bearer ${API_KEY}` } }); const data = await res.json(); // data.compras = [{ data, produto, valor, nf }] // Calcular fase CRM automaticamente pela última compra const dias = calcularDias(data.compras[0].data); const fase = dias > 730 ? "Resgate" : dias > 365 ? "Receita Vencida" : dias > 75 ? "Ativação" : "Pós-Venda"; return { compras: data.compras, faseCRM: fase }; }
4. Sincronização Automática a cada 30 minutos
async function sincronizarCRM() { const [clientes, vencidas] = await Promise.all([ fetch(`${API_BASE}/clientes`).then(r => r.json()), receitasVencidas() ]); clientes.forEach(c => { c.faseCRM = calcularFase(c.ultimaCompra); c.receitaVencida = vencidas.some(v => v.cpf === c.cpf); }); atualizarInterface(clientes); // re-render automático } // Roda ao abrir o sistema e a cada 30 minutos sincronizarCRM(); setInterval(sincronizarCRM, 30 * 60 * 1000);
